	問2
	アクセス記録テーブル(access_log)からリンク元URLを重複のない形式で取り出してみましょう
	

	問題	
		[①]
			referer	
		[②]
			access_log
		;
		
		

	解答	
		
		[①]SELECT DISTINCT
		[②]FROM
		
		
	実行文
	
		SELECT	DISTINCT	--なんの情報が欲しいのか重複せず一列で選択させる
			referer			--regerer列を取り出す命令
		FROM				--どこのテーブルから情報を取り出すのか選択させる
			access_log		--access_logから情報をとることを宣言
		;					--命令を終了させる
