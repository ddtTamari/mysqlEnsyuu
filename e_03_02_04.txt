	問4
	employeeに対して、表のデータを挿入するためのSQL命令ですが、2点誤りを指摘しなさい。
										
	問題	
		INSERT INTO
			employee
			(
			 s_id
			 l_name
			 depart_id
			 last_update
			 )
		VALUE
			(
			 'WA00001',
			 '和田',
			 'J01',
			 CURRENT_DATE
			 )
		;
		
	解答
		
		INSERT INTO
			employee
			(
			 s_id,
			 l_name,
			 depart_id,
			 last_update
			 )
		VALUES
			(
			 'WA00001',
			 '和田',
			 'J01',
			 CURRENT_DATE
			 )
		;
			
	実行文
	
		INSERT INtO                                   -- データを挿入するテーブル、列を指定する
			employee                                  -- employeeテーブルに挿入する
			(                                         -- どこの列に挿入するか指定を始めるカッコ
				s_id,                                 -- 社員コード列に挿入する
				l_name,                               -- 社員名(氏)列に挿入する
				depart_id,                            -- 所属部署コード列に挿入する
				last_update                           -- 最終更新日列に挿入する
			 )                                        -- どの列に挿入するか指定の終了を表すカッコ
		VALUES                                        -- 挿入するデータを指定する
			(                                         -- 入力データの始まりを指定するカッコ
			 'WA00001',                               -- 社員コードを挿入する
			 '和田',                                  -- 社員名(氏)を挿入する
			 'J01',                                   -- 所属部署コードを挿入する
			 CURRENT_DATE                             -- 今日の日付を挿入する
			)                                         -- 入力データの終わりを意味するカッコ
		;                                             -- 命令を終了させる
