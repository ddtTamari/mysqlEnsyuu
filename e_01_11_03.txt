	問3
	usrから利用者と完全な住所を利用者コードについて降順で取り出すためのSQL命令を記述
	別名は「利用者名」「住所」
	
	問題	
		
	解答
		SELECT
			CONCAT(l_name,f_name) AS 利用者名,
			CONCAT(prefecture,city,o_address) AS 住所
		FROM
			usr
		ORDER BY
			user_id  DESC
		;
	実行文
	
		SELECT														--なんの情報が欲しいのか選択させる
			CONCAT(l_name,f_name) AS 利用者名,						--姓と名を結合して利用者名と列名を変えを取り出す
			CONCAT(prefecture,city,o_address) AS 住所				--住所の情報を結合し住所として列名を変え取り出す
		FROM														--どこのテーブルから情報を取り出すのか選択させる
			usr														--usrから情報をとることを宣言
		ORDER BY													--並び替えを設定する
			user_id  DESC											--利用者コードについて降順で表示させる
			;														--命令を終了させる
