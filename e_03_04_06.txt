	問6
	employeeにおいて、役職列が部長である場合は「リーダ」にある場合には「マネージャ」に名称変更しましょう
	
	問題
		UPDATE
			employee
		SET
			class =
			(
			[①]
				class
					[②] '部長' THEN 'リーダ'
					[②] '課長' THEN 'マネージャ'
					ELSE [③]
			END
			)
		;
		
	解答
		[①] CASE
		[②] WHEN
		[③]class
		
	実行文
	
	UPDATE                                        -- 更新を行う
		employee                                  -- employeeテーブルの更新を行う
	SET                                           -- 更新する値を指定する
		class =                                   -- 役職列に代入する
		(                                         -- 条件をくくるカッコ
		CASE                                      -- 条件によって求める値を変える
			class                                 -- 役職列が以下の条件のとき
				WHEN '部長' THEN 'リーダ'         -- 部長という役職ならリーダに置き換え
				WHEN '課長' THEN 'マネージャ'     -- 課長という役職ならマネージャに置き換え
				ELSE class                        -- 上記意外ならそのままの値
		END                                       -- 条件を終わらせる
		)                                         -- 条件をくくるカッコ
		;                                         -- 命令を終了させる
		
		実行結果表示命令文
		
		SELECT                -- どの列を表示させたいか指定する
			*                 -- すべての列を表示させる
		FROM                  -- どこのテーブルから列を取り出すのか
			employee          -- からデータを取り出す
		;
