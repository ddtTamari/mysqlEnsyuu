		VALUES                                        -- 挿入するデータを指定する
			(                                         -- 入力データの始まりを指定するカッコ
			 '佐々木三郎',                            -- 名前を挿入する

			 '興味深い内容です。',                    -- 答え2の内容を挿入する

			)                                         -- 入力データの終わりを意味するカッコ
		;                                             -- 命令を終了させる
