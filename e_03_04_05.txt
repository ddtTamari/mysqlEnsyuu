	問5
	questに「男性」「女性」「男」「女」と表記の揺らぎがありましたので、これを「男」「女」という表記に統一したいと思います。以下のSQL命令に２点誤りがあります、指摘しなさい。
	問題
		
		UPDATE INTO
			quest
		VALUES
			sex = REPLACE(sex,'性','')
		;
		
	解答
		①UPDATE INTOになっていたので、不要なINTOを削除
		②VALUESになっていたので、更新はSETなのでSETに変更
		
		UPDATE
			quest
		SET
			sex = REPLACE(sex,'性','')
		;
		
	実行文
		
		UPDATE                                     -- 更新するテーブルを指定する
			quest                                  -- questテーブルを更新する
		SET                                        -- 更新する値を指定する
			sex = REPLACE(sex,'性','')             -- 性別列にある性の字を何もない状態にして置き換える
		;                                          -- 命令を終了させる
