	問5
	rental上利用者コードがB200405,ISBNコードが4－0010－0000－1のデータについて、貸出日を今日の日付に更新するためのSQL命令文の誤りを2点指摘しなさい。
	
	問題	
		UPDATE
			rental
		SET
			rental_date CURRENT_DATE
		WHERE
			user_id = 'B200405'
		   OR
		    isbn = '4-0010-0000-1'
		;
		
	解答
		①今日の日付に更新する文に"="が抜けていたので追加しました
		②条件文にまたはを意味するORになっていましたが正確にはかつのANDなのでANDに変更しました
		
		UPDATE
			rental
		SET
			rental_date = CURRENT_DATE
		WHERE
			user_id = 'B200405'
		   AND
		    isbn = '4-0010-0000-1'
		;
	実行文
		UPDATE                                    -- 更新するテーブルを指定する
			rental                                -- rentalテーブルを更新する
		SET                                       -- 更新する値を指定する
			rental_date = CURRENT_DATE            -- 貸出日を今日の日付に変更
		WHERE                                     -- 更新するデータに条件を追加
			user_id = 'B200405'                   -- 利用者コードがB200405のデータを更新
		   AND                                    -- かつ
		    isbn = '4-0010-0000-1'                -- isbnコードが4-0010-0000-1のデータを更新する
		;                                         -- 命令を終了させる
		
		実行結果表示命令文
		
		SELECT                -- どの列を表示させたいか指定する
			*                 -- すべての列を表示させる
		FROM                  -- どこのテーブルから列を取り出すのか
			rental            -- rentalからデータを取り出す
		;                     -- 命令を終了させる
